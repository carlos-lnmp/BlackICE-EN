#Ghazi I Death

country_event = {
	id = iraq.4
	title = iraq.4.t 
	desc = iraq.4.d
	picture = GFX_report_event_king_gfxaddon #GFX_country_event_king_ghazi
	
	fire_only_once = yes
	is_triggered_only = yes
	trigger = {
		tag = IRQ
			check_variable = { global.days_passed > 1189 } # 1939-04-04
	}
	
	option = { # Abd al-Ilah
		name = iraq.4.a
		kill_country_leader = yes
		set_country_flag = ghazi_dead
		create_country_leader = {
			name = "Abd al-Ilah"
			desc = "POLITICS_ABD_DESC"
			picture = "Portrait_Abd_al-Ilah.dds"  ### maybe currently working?
			expire = "1965.1.1"
			ideology = emiratism
			traits = {
				monarchist
			}
		}
		swap_ideas = {
			remove_idea = IRQ_ghazi_faisal
			add_idea = IRQ_abd_al-ilah
		}
		add_popularity = {
				ideology = monarchism
				popularity = 0.05
		}
	}
}
